\section{Repaso de cápsula convexa (convex hull)}

\begin{frame}{Cápsula convexa}

    \begin{itemize}
        \item La forma que adopta una banda elástica, si la hacemos rodear ``postes'' clavados en los puntos.
        \item Se puede calcular con Graham Scan, en un solo recorrido.
        \item Se puede calcular en dos partes: La \textbf{cápsula superior} y la \textbf{cápsula inferior}.
        \item Ambos algoritmos se basan en ``tener un stack e ir sacando hasta que encaje''.

    \end{itemize}

\end{frame}

\section{Dualidad punto línea}

\begin{frame}{Representación de rectas en computadora}

    \begin{itemize}
        \item Idea: ¿Cómo representamos una recta?
        \pause
        \invisible<1>{
                \item Dependiendo del contexto y de cómo la vamos a usar hay muchas maneras.
                \item Para ``cálculos geométricos con rectas y figuras'', en general representaciones con \textbf{vectores} es lo mejor.
            }
        \pause
        \invisible<1-2>{\item Si la recta representa naturalmente una cierta \textit{función lineal} que a veces evaluamos, la \textbf{forma explícita} $y = ax + b$ puede ser más útil.
                      \item Notar que de ser así no tienen sentido las rectas verticales (no son funciones).
                      \item Salvo cuando aclaremos lo contrario, ``recta'' para nosotros será en particular durante toda esta charla ``recta no vertical''.
        }
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Rectas y puntos en struct}

\end{verbatim}

\begin{itemize}
        \pause
        \invisible<1>{\item ¿Qué diferencia hay entre ambas?

\end{itemize}
    
\end{frame}

\begin{frame}{La transformación}

    \begin{itemize}
        \item Transformar rectas en puntos: $y = ax + b \mapsto (-a,-b)$ 
        \item Transformar puntos en rectas: $(a,b) \mapsto y = -ax -b$ 
        \item En términos de los structs anteriores, siempre es $(f_1, f_2) \mapsto (-f_1, -f_2)$, más allá de los nombres de los \textit{fields} $f_1$, $f_2$ en el struct que corresponda.
        \pause
        \invisible<1>{
                \item ¿Eso es todo? ¿Eso es dualidad punto recta?
            }
        \pause
        \invisible<1-2>{\item Sí.
        }
        \pause
        \invisible<1-3>{\item ¿De verdad?
        }
        \pause
        \invisible<1-4>{\item Sí.}
        \pause
        \invisible<1-5>{\item Pero para que sea útil hay que entender la \textbf{relación} entre ellos.
        }
    \end{itemize}
    
\end{frame}

\begin{frame}{Relaciones}

    \begin{itemize}
        \item \textbf{No} transforma el plano como una rotación/traslación/reflexión:\\ \textbf{puntos no van a puntos}.
        \item Transforma puntos y rectas, e \textbf{intercambia sus roles}.
        \item Por esta razón, una forma de ilustrar la transformación es mediante colores:
        
        \includegraphics[scale=0.09]{fig1.png}        $\ $
        \includegraphics[scale=0.09]{fig2.png}
        
        Aproximadamente:
        
        \item A) $f(x) = y = -x - 5$ $\leftrightarrow$ $p = (1,5)$
        
        \item B) $f(x) = y = 1$ $\leftrightarrow$ $p = (0,-1)$
        
        \item C) $f(x) = y = x - 5$ $\leftrightarrow$ $p = (-1,5)$
    \end{itemize}
\end{frame}

\begin{frame}{Relaciones}

    \begin{itemize}
        
        \item $y = f(x) = ax + b$, y $(a,b) = p$, la dualidad intercambia $p$ y $f$
        \item Al dualizar dos veces volvemos al mismo punto / recta de partida
        \pause
        \invisible<1>{
                \item ``rectas no verticales'' = `` funciones lineales'' $\Leftrightarrow$ ``puntos del plano''
            }
        \pause
        \invisible<1-2>{\item ``pendiente mayor'' $\Leftrightarrow$ ``coordenada $x$ menor''
        }
        \pause
        \invisible<1-3>{\item ``rectas paralelas'' $\Leftrightarrow$ ``alineados en vertical''
        }
    \end{itemize}
    
\end{frame}

\begin{frame}{Operador está arriba/abajo}

    \begin{itemize}
        
        \item Si $p = (x_p,y_p)$, $R$ es $y = ax + b$, decimos $p \uparrow R$ si $y_p \geq ax_p + b$
        \item Lo leemos `` $p$ está arriba de $R$ ''
        \item Análogamente $p \downarrow R$
        \item Observación: $p \in R \Leftrightarrow p \uparrow R \land p \downarrow R$
    \end{itemize}
    
\end{frame}

\begin{frame}{Relación esencial de la dualidad punto-línea}

    \begin{itemize}
        \item $p \uparrow R \Leftrightarrow dual(R) \uparrow dual(p)$
        \item Corolario: $p \in R \Leftrightarrow dual(R) \in dual(p)$
        \pause
        \invisible<1>{
            \item Demostración:
                \begin{itemize}
                    \item $p \uparrow R$ es por definición  $y_p \geq a x_p + b$
                    \item $dual(R) \uparrow dual(p)$ es por definición $-b \geq (-x_p)(-a) + (-y_p)$
                    \item Ambas condiciones son equivalentes
                    \item Nota: esta es la razón para poner los signos negativos en la transformación
                \end{itemize}
        }
    \end{itemize}
    
\end{frame}

\begin{frame}{Relaciones de incidencia}

    De lo anterior tenemos:

    \begin{itemize}
        \item ``muchos puntos arriba de una recta'' $\Leftrightarrow$ \\``muchas rectas que pasan por debajo de un punto''
        \item ``rectas concurrentes'' $\Leftrightarrow$ ``puntos alineados (pero no en vertical)''
    \end{itemize}
    
\end{frame}


\begin{frame}{Relaciones de extremos}

    \begin{itemize}
        \item ``$y$'' (o sea $f(x)$ dado $x$) $\Leftrightarrow$ $p \cdot (-x,-1)$
        \pause
        \invisible<1>{\item ``Recta con mayor $f(x)$'' $\Leftrightarrow$ ``Punto que está más avanzado en la dirección apuntada por $\vec v=(-x,-1)$''.
        }
        \pause
        \invisible<1-2>{\item \textbf{``Envolvente superior'' $\Leftrightarrow$ ``Cápsula inferior''.}
                        \item \textbf{``Envolvente inferior'' $\Leftrightarrow$ ``Cápsula superior''.}
        }
        \pause
        \invisible<1-3>{\item Esta última es la que da el nombre a ``convex hull trick''
                        \item ¡Revisar los problemas anteriores para verificar los emparejamientos!}

    \end{itemize}
    
\end{frame}

\section{Convex Hull Trick}

\begin{frame}{Qué es}

    \begin{itemize}
        \invisible<0-0>{\item Es la técnica que usamos para calcular eficientemente,a partir de un $x$, el valor de  $max_{i=1}^n {f_i(x)}$, si tenemos $n$
                             funciones lineales $f_i$}
        \pause
        \invisible<1-1>{\item Permite mantener la envolvente superior de rectas eficientemente.}
        \pause
        \invisible<1-2>{\item Por dualidad, \textbf{es equivalente tener la cápsula inferior de puntos}, que son los únicos puntos candidatos a ser los extremos en una dirección $\vec v=(-x,-1)$.}
    \end{itemize}
    
\end{frame}

\begin{frame}{Convex Hull Trick offline}

    \begin{itemize}
        \item Si las rectas van apareciendo \textbf{en orden de pendiente}, o si las podemos ordenar libremente, hacemos lo mismo que para construir cápsula inferior/superior si los puntos están ordenados por $x$
        \item Ante cada nueva recta, vamos ``descartando hasta que encaje'' del vector de rectas, y siempre tendremos las rectas de la envolvente \textbf{ordenadas} (binary search).
    \end{itemize}
    
\end{frame}


\begin{frame}{Convex Hull Trick online}
    \begin{itemize}
        \invisible<0-0>{\item Si las rectas aparecen en forma dinámica y en cualquier orden, podemos tenerlas en un \texttt{set} \textbf{ordenadas por pendiente}, y usar \texttt{lower\_bound} para ver dónde insertar la nueva (si es que hay que hacerlo).}
        \pause
        \invisible<1-1>{\item Podemos usar la dualidad para razonar más fácil en términos de puntos (si así nos resulta más fácil), pues es análogo a mantener la cápsula inferior cuando aparecen nuevos puntos: allí los mantendríamos ordenados por coordenada $x$, y luego ``borraríamos adyacentes a cada lado'' hasta que la banda elástica quede ``correctamente tensada''.}
        \pause
        \invisible<1-2>{\item O más aún podemos programar directamente el código para mantener la cápsula inferior, y traducir (dualizar) los parámetros de las queries.}
    \end{itemize}
    
\end{frame}

\begin{frame}{Convex Hull Trick online (implementación)}
    \begin{itemize}
        \item La implementación es muy delicada:
            \begin{itemize}
                \item Para las inserciones, queremos hacer lower-bound por (pendiente de recta / $x$ del punto)
                \item Para las queries, queremos hacer lower-bound por (``derivada'' de $f(x)$ / derivada de $p \cdot (-x, -1)$)
                \item Si bien el valor que damos es distinto, con ambos criterios el conjunto está bien ordenado
                \item Hack para set de C++: Se puede tener un comparator que cambia su comportamiento según variables globales
            \end{itemize}
        \item Recomendación: sentarse con tiempo y mucho cuidado a entender la idea e implementarla, y guardar la implementación para usar en competencias
        
    \end{itemize}
    
\end{frame}

\section{Dualidad con rectas verticales (Opcional Bonus)}

\begin{frame}{Nos faltan puntos}
    \begin{itemize}
        \item Tendríamos que poder dualizar las rectas verticales, pero ya usamos todos los puntos del plano.
        \item Solución: \textbf{Agregar} puntos al plano.
        \item ¿Dónde tendría que estar el punto dual de una recta vertical?
    \end{itemize}
    
\end{frame}

\begin{frame}{Nos faltan puntos (cont.)}
    \begin{itemize}
        \item Por el punto $dual(R)$, pasan todas las rectas $dual(p)$ para los puntos $p \in R$.
        \item ¡Si $R$ es vertical, esos puntos dualizan a rectas paralelas!
        \item Necesitamos agregar puntos en el infinito, donde las rectas paralelas se cortan.
        \item Por cada posible pendiente de rectas paralelas, habrá un punto en el infinito.
        \item La recta vertical $x = a$ dualiza al punto infinito donde convergen las rectas de pendiente $-a$.
        \item ¿Dónde se cortan las rectas verticales? 
    \end{itemize}
\end{frame}

\begin{frame}{Nos faltan puntos (cont.)}
    \begin{itemize}
        \item Se cortan en otro punto en el infinito, que no tiene todavía recta dual...
        \item La recta dual del ``infinito vertical'' es una ``recta en el infinito'', que pasa exactamente por todos los puntos infinitos.
    \end{itemize}
\end{frame}

\begin{frame}{Plano proyectivo}
    \begin{itemize}
        \item Lo anterior hace que toda recta y todo punto tenga dual, y mantiene la relación esencial de dualidad punto línea.
        \item El plano con los puntos y rectas que agregamos se llama plano proyectivo.
        \item Una forma equivalente de construirlo (pero con un cambio de coordenadas) sin separar como especial la recta ``infinita'' es pensar que tomamos:
            \begin{itemize}
                \item Los subespacios de dimensión 1 en $\mathbb{R}^3$, como puntos
                \item Los subespacios de dimensión 2 en $\mathbb{R}^3$, como rectas
                \item Dualizar es tomar complemento ortogonal
            \end{itemize}
        \item En general, estos conceptos no se usan en programación competitiva :)
    \end{itemize}
\end{frame}

\begin{frame}{Un ejemplo: Problema el $K$ en el plano proyectivo}

    Los siguientes problemas son duales (al considerarlo en el plano proyectivo)

    \begin{itemize}
        \item Dados dos \textbf{puntos} y $N$ \textbf{rectas}, cuántas \textbf{rectas} hay que cruzar como mínimo, al mover \textbf{un punto} hasta la posición del otro, pero se puede ``dar la vuelta por el infinito''.
        \item Dadas dos \textbf{rectas} y $N$ \textbf{puntos}, cuántos \textbf{puntos} hay que cruzar como mínimo, al mover \textbf{una recta} hasta la posición de la otra. 
    \end{itemize}

    En este caso, la ``circularidad'' que complica el problema natural de rectas, se vuelve explícita al dualizar.

\end{frame}
